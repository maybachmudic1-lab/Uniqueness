      document.querySelectorAll('.smart-tg-link').forEach(link => {
        link.addEventListener('click', e => {
          e.preventDefault();
          tgInfoPopup.style.display = 'flex';
        });
      });
      
      const closeTgPopup = () => tgInfoPopup.style.display = 'none';
      closeTgPopupBtn.addEventListener('click', closeTgPopup);
      tgInfoPopup.addEventListener('click', e => { if (e.target === tgInfoPopup) closeTgPopup(); });

      copyTgBtn.addEventListener('click', () => {
        navigator.clipboard.writeText(tgUsernameInput.value).then(() => {
          copyTgBtn.textContent = "Copied!";
          copyTgBtn.classList.add("copied");
          setTimeout(() => { copyTgBtn.textContent = "Copy"; copyTgBtn.classList.remove("copied"); }, 2000);
        });
      });
      
      // --- Mentor Popup Logic (Timed) ---
      const mentorPopup = document.getElementById('mentor-popup');
      const closeMentorPopupBtn = document.getElementById('close-mentor-popup');
      
      // ‚úÖ Show the popup after 5 seconds (5000 milliseconds)
      setTimeout(() => {
        mentorPopup.style.display = 'flex';
      }, 5000);
      
      const closeMentorPopup = () => mentorPopup.style.display = 'none';
      closeMentorPopupBtn.addEventListener('click', closeMentorPopup);
      mentorPopup.addEventListener('click', e => { if (e.target === mentorPopup) closeMentorPopup(); });

      // Modules logic
      const container=document.getElementById('modules-container');
      const prevBtn=document.getElementById('prevBtn');
      const nextBtn=document.getElementById('nextBtn');
      let modules=[],currentIndex=0;

      fetch('modules.json')
        .then(res => {
            if (!res.ok) {
                throw new Error(`HTTP error! status: ${res.status}`);
            }
            return res.json();
        })
        .then(data => {
            modules = data;
            showModule(0);
        })
        .catch(err => {
            container.textContent = "‚ö†Ô∏è Error loading modules. Please check if modules.json is available and correctly formatted.";
            console.error(err);
        });

      function showModule(index){
        if(index<0||index>=modules.length) return;
        currentIndex=index;
        container.innerHTML=`
          <div class="module-card">
            <div class="module-title">${modules[index].title}</div>
            <div class="module-content">${modules[index].content}</div>
          </div>`;
        prevBtn.disabled=index===0;
        nextBtn.disabled=index===modules.length-1;
      }
      prevBtn.addEventListener('click',()=>showModule(currentIndex-1));
      nextBtn.addEventListener('click',()=>showModule(currentIndex+1));
    });
